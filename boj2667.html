n = int(input())  # 지도의 크기 (n * n)
apart = []        # 지도 데이터
ans = [0] * 1001  # 각 단지의 아파트 수 (일단 모두 0으로 초기화 후 나중에 0인 데이터는 출력하지 않음)
danji = 0         # 단지의 개수
 
# 상하좌우 탐색 함수 시작
 
def updownleftright(x, y):
 
    # 한칸 오른쪽
    if x != n-1:                      # index 범위 벗어남 에러를 막기 위해서 조건 설정
        if apart[x+1][y] == 1:        # 한칸 오른쪽의 인덱스에 아파트가 있다면
            apart[x+1][y] = 0         # 0으로 표시해서 나중에 방문하지 않도록 함
            ans[danji] += 1           # 해당 단지의 아파트 수를 1 늘려줌
            updownleftright(x+1, y)   # 한칸 오른쪽으로 다시 함수 실행
 
    # 한칸 왼쪽
    if x != 0:
        if apart[x-1][y] == 1:
            apart[x-1][y] = 0
            ans[danji] += 1
            updownleftright(x-1, y)
 
    # 한칸 위쪽
    if y != n-1:
        if apart[x][y+1] == 1:
            apart[x][y+1] = 0
            ans[danji] += 1
            updownleftright(x, y+1)
 
    # 한칸 아래쪽
    if y != 0:
        if apart[x][y-1] == 1:
            apart[x][y-1] = 0
            ans[danji] += 1
            updownleftright(x, y-1)
 
# 상하좌우 탐색 함수 종료
 
# 입력
for i in range(n):
    apart.append(list(map(int, input()))) # 지도 데이터 입력(각각 int형으로 map 해준 것을 list로 넣어줘서 list안에 list가 쌓여 2차원 배열을 만듬)
 
# 연산
for i in range(n):                        
    for j in range(n):                    
        if apart[i][j] == 1:              # 만약 지도에서 i, j 인덱스에 아파트가 있다면,
            danji += 1                    # 단지 수를 1 늘려줌
            ans[danji] += 1               # 해당 단지의 아파트 수를 1 늘려줌
            apart[i][j] == 0              # 이미 방문했기 때문에  0으로 표시해서 방문하지 않도록 함
            updownleftright(i, j)         # 상하좌우 탐색
 
# 출력
print(danji)                              # 총 단지 개수 출력
for a in sorted(ans):                     # 각 단지의 수를 정렬해서 반복
    if a == 1: print(a)                   # a가 1일경우 그냥 a 출력 (만약 단지에 아파트가 1개일 경우상하좌우 탐색 함수에서 if문에 충족하지 않으므로 상하좌우 탐색 함수의 +1을 받지 못했기 때문에)
    if a > 1: print(a-1)                  # a가 1보다 클 경우 a-1 출력 (상하좌우 탐색 함수에서 +1을 또 받았기 때문에)
